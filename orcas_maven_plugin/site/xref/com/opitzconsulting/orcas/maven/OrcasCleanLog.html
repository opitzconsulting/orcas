3   import java.io.File;
4   
5   import org.apache.maven.plugins.annotations.Mojo;
6   
7   import de.opitzconsulting.orcas.diff.ParametersCall;
8   import de.opitzconsulting.orcas.diff.Orcas;
9   
10  /**
11   * Cleans the spoolfolder.
12   */
13  @Mojo( name = "cleanLog" )
14  public class OrcasCleanLog extends BaseOrcasMojo
15  {
16    @Override
17    protected void executeWithParameters( ParametersCall pParameters )
18    {
19      if( pParameters.getSpoolfolder().length() != 0 && new File( pParameters.getSpoolfolder() ).exists() )
20      {
21        Orcas.deleteRecursive( new File( pParameters.getSpoolfolder() ) );
22      }
23    }
24  
25    @Override
26    protected String getLogname()
27    {
28      return "clean-log";
29    }
30  }
